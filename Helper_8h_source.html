92 static T weightedArrayElement(const std::vector<WeightedElement<T>>& data)
93 {
94 if (data.empty())
95 {
96 throw std::invalid_argument{"Data is empty."};
97 }
98
99 const auto sumOfWeights =
100 std::accumulate(data.begin(), data.end(), 0u,
101 [](unsigned sum, const WeightedElement<T>& element) { return sum + element.weight; });
102
103 if (sumOfWeights == 0u)
104 {
105 throw std::invalid_argument{"Sum of weights is zero."};
106 }
107
108 const std::integral auto targetWeightValue = Number::integer<unsigned>(sumOfWeights);
109
110 unsigned currentSum = 0;
111
112 for (const auto& element : data)
113 {
114 currentSum += element.weight;
115
116 if (targetWeightValue <= currentSum)
117 {
118 return element.value;
119 }
120 }
121
122 return data.at(data.size() - 1).value;
123 }
